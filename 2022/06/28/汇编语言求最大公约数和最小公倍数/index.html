汇编语言求最大公约数和最小公倍数

发表于 2022-06-28 分类于 汇编语言 阅读次数： 本文字数： 1.8k 阅读时长 ≈ 13 分钟

DATA SEGMENT
NUMBER1 DW ?
NUMBER2 DW ?
MAX DW ?
MIN DW ?
TIP1 DB ‘Input Number1: ‘
TIP2 DB ‘Input Number2: ‘
COUNT EQU $-TIP2
MAXYUE DB ‘Max Gong yue : ‘
MINBEI DB ‘Min Gong bei : ‘
QT DB ‘Press q To Exit’
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START: MOV AX,DATA
MOV DS,AX

AGAIN: MOV CX, COUNT
MOV DI, OFFSET TIP1
MOV AH, 2
L1: MOV DL, [DI]
INT 21H
INC DI
LOOP L1 ;显示’Input Number1: ‘

        CALL INPUT    
        MOV NUMBER1, BX
        
        MOV CX, COUNT
        MOV DI, OFFSET TIP2
        MOV    AH, 2
    L2:    MOV DL, [DI]
        INT 21H
        INC DI
        LOOP L2        ;显示'Input Number2: '
        
        CALL INPUT          
        MOV NUMBER2, BX   
        
        MOV DI, OFFSET MAXYUE
        MOV CX, COUNT
        CALL YUE
        
        MOV CX, COUNT
        CALL BEI
        
        MOV AH, 2
        MOV DL, 0AH
        INT 21H
        MOV DL, 0DH
        INT 21H
        
        MOV DI, OFFSET QT
        MOV CX, COUNT 
    L3:    MOV DL, [DI]
        INT 21H
        INC DI
        LOOP L3        ;显示'Press q To Exit'
        
        MOV AH,1
        INT 21H
        CMP AL, 'q'
        JZ QUIT
        MOV AH, 2
        MOV DL, 0AH 
        INT 21H
        MOV DL, 0DH 
        INT 21H
        MOV DL, 0AH 
        INT 21H         ;这一小段是主函数
        JMP AGAIN       ;通过CALL调用下面的程序

;———————————————————————
INPUT: MOV BX, 0;整个INPUT的功能是从键盘输入正整数存到BX
NEXT: MOV AH, 1;NEXT指的是输入下一位上的数
INT 21H
CMP AL, 0DH ; 输入回车时结束
JZ BREAK
AND AX, 000FH;将AL中的ASCII码转换为数字,比如’2’=32H=110010B,AND完变成10B=2(十进制)
XCHG AX, BX
MOV CX, 10
MUL CX
ADD BX, AX;之前输入的数乘以10 加上新输入的数
JMP NEXT
BREAK: RET

DISPLAY: MOV AH, 2 ;DISPALY的功能是
MOV DL, 0AH ;回车换行后,以十进制显示BX当中的数
INT 21H
MOV DL, 0DH
INT 21H
L4: MOV DL, [DI] ;第一次调用显示’Max Gong yue : ‘
INT 21H ;第二次调用显示’Min Gong bei : ‘
INC DI
LOOP L4
MOV AX, BX
MOV BX, 10
MOV CX, 0
PART1: MOV DX, 0;把要显示的数除以10,然后把余数入栈
INC CX ;记录入栈个数
DIV BX
PUSH DX;如果除数为16位,则结果的商会存放在AX中,余数会存放在DX中。
CMP AX, 0 ;余数为0时结束
JNZ PART1
;整个PART1的功能是把十进制数从个位开始,把每个位上的数入栈
PART2: POP AX
ADD AL, 30H ;转成可以显示的ASCII码
MOV DL, AL
MOV AH, 2
INT 21H
LOOP PART2
RET
;整个PART2的功能是把PART1存入栈中的数出栈并显示在屏幕上

YUE: MOV BX, 1 ;求NUMBER1和NUMBER2的最大公约数,BX当作除数
P1: MOV DX, 0
MOV AX, NUMBER1
DIV BX
CMP DX, 0
JNZ P2 ; 如果BX不能被NUMBER1整除,BX一定不是公约数,跳到P2
MOV DX, 0
MOV AX, NUMBER2
DIV BX
CMP DX, 0
JNZ P2 ; 如果BX不能被NUMBER2整除,BX一定不是公约数,跳到P2
MOV MAX, BX ;既能被NUMBER1整除又能被NUMBER2整除,把当前BX放到MAX里面
P2: CMP BX, NUMBER1;BX等于其中一个数(如NUMBER1)的时候退出
INC BX ;判断BX+1是不是公约数
JNZ P1
MOV BX, MAX
CALL DISPLAY;显示MAX
RET

BEI: MOV AX, NUMBER1;最大公倍数就是两数相乘后除以最大公约数
MUL NUMBER2
DIV MAX
MOV MIN, AX
MOV BX, MIN
CALL DISPLAY;显示MIN
RET

QUIT: MOV AH, 4CH
INT 21H
CODE ENDS
END START
